USE WebStor

-- Завдання 1 --

SELECT EMPL_NUM,NAME, AGE,REP_OFFICE
FROM SALESREPS
WHERE AGE > 45
ORDER BY AGE DESC

-- Завдання 2 --

SELECT DISTINCT MFR , PRODUCT 
FROM ORDERS
WHERE YEAR(ORDER_DATE) = 2008

-- Завдання 3 -- 

SELECT TOP 1 WITH TIES REP AS ID, COUNT(*) AS ORDER_COUNT
FROM ORDERS
WHERE YEAR(ORDER_DATE) = 2008
GROUP BY REP
ORDER BY ORDER_COUNT DESC
